// ePhone — головний файл прошивки
#include
#include
#include "screens/BootScreen.h"
#include "screens/MenuScreen.h"
#include "services/TouchManager.h"

void setup() {
  Serial.begin(115200);
  lv_init();
  tft.begin();
  touch.begin();
  BootScreen::show();
  MenuScreen::show();
}

void loop() {
  lv_timer_handler();
  touch.update();
  delay(5);
}
